gateway = SentinelGateway(config)

result = gateway.invoke(
    "User input here",
    agent_executor=lambda x: "Agent response"
)

print(f"Risk: {result['aggregated_risk']['overall_risk_score']:.2f}")
print(f"Blocked: {result['blocked']}")
